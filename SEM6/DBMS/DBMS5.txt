CREATE OR REPLACE TRIGGER INSERT_TRIGGER
AFTER INSERT OR UPDATE OF MARKS ON RESULT 
FOR EACH ROW
BEGIN
IF :NEW.MARKS>=50 THEN
DELETE FROM BACKPAPER WHERE ROLL=:NEW.ROLL AND SCODE=:NEW.SCODE;
ELSE 
INSERT INTO BACKPAPER VALUES(:NEW.ROLL,:NEW.SCODE);
END IF;
END;

------------------------------------------------
CREATE OR REPLACE TRIGGER DELETE_TRIGGER 
before DELETE ON REQUEST 
FOR EACH ROW
DECLARE 
CURR_CNT SERVICE_LOG.ENTRY_NO%TYPE;
BEGIN
SELECT COUNT(*) INTO CURR_CNT FROM SERVICE_LOG;
CURR_CNT:=CURR_CNT +1;
INSERT INTO SERVICE_LOG VALUES(CURR_CNT,:OLD.REQ_NO,SYSDATE);
END;
